-- [[ Funções de Hitbox ]]
local originalSizes = {}

local function saveOriginalSize(part)
    if not originalSizes[part] then
        originalSizes[part] = part.Size
    end
end

local function applyHitbox(character, color)
    local hrp = character:FindFirstChild("HumanoidRootPart")
    local head = character:FindFirstChild("Head")

    -- Body Hitbox
    if hrp and getgenv().BodyHitbox then
        hrp.Size = Vector3.new(getgenv().HitboxSize, getgenv().HitboxSize, getgenv().HitboxSize)
        hrp.Transparency = getgenv().HitboxTransparency
        hrp.BrickColor = BrickColor.new(color)
        hrp.Material = "Neon"
        hrp.CanCollide = false
    elseif hrp and not getgenv().BodyHitbox then
        hrp.Size = Vector3.new(2, 2, 1)
        hrp.Transparency = 1
        hrp.BrickColor = BrickColor.new("Medium stone grey")
        hrp.Material = "Plastic"
        hrp.CanCollide = false
    end

    -- Head Hitbox
    if head and getgenv().HeadHitbox then
        head.Size = Vector3.new(getgenv().HeadHitboxSize, getgenv().HeadHitboxSize, getgenv().HeadHitboxSize)
        head.Transparency = getgenv().HitboxTransparency
        head.BrickColor = BrickColor.new(color)
        head.Material = "Neon"
        head.CanCollide = false
    elseif head and not getgenv().HeadHitbox then
        head.Size = Vector3.new(1, 1, 1)
        head.Transparency = 1
        head.BrickColor = BrickColor.new("Medium stone grey")
        head.Material = "Plastic"
        head.CanCollide = false
    end

    -- Melee Hitbox (expande todas as partes do corpo)
    if getgenv().MeleeHitbox then
        for _, part in next, character:GetDescendants() do
            if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" and part.Name ~= "Head" then
                pcall(function()
                    saveOriginalSize(part)
                    local s = getgenv().MeleeSize
                    part.Size = originalSizes[part] + Vector3.new(s, s, s)
                    part.Transparency = getgenv().HitboxTransparency
                    part.CanCollide = false
                end)
            end
        end
    else
        -- Reset partes do melee
        for _, part in next, character:GetDescendants() do
            if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" and part.Name ~= "Head" then
                pcall(function()
                    if originalSizes[part] then
                        part.Size = originalSizes[part]
                        part.Transparency = 0
                        part.CanCollide = true
                    end
                end)
            end
        end
    end
end

local function resetHitbox(character)
    local hrp = character:FindFirstChild("HumanoidRootPart")
    local head = character:FindFirstChild("Head")
    if hrp then
        hrp.Size = Vector3.new(2, 2, 1)
        hrp.Transparency = 1
        hrp.BrickColor = BrickColor.new("Medium stone grey")
        hrp.Material = "Plastic"
        hrp.CanCollide = false
    end
    if head then
        head.Size = Vector3.new(1, 1, 1)
        head.Transparency = 1
        head.BrickColor = BrickColor.new("Medium stone grey")
        head.Material = "Plastic"
        head.CanCollide = false
    end
    -- Reset melee parts
    for _, part in next, character:GetDescendants() do
        if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" and part.Name ~= "Head" then
            pcall(function()
                if originalSizes[part] then
                    part.Size = originalSizes[part]
                    part.Transparency = 0
                    part.CanCollide = true
                end
            end)
        end
    end
end

local function getPlayerChars()
    local chars = {}
    for _, p in next, Players:GetPlayers() do
        if p.Character then chars[p.Character] = true end
    end
    return chars
end

-- [[ Loop Principal do Hitbox ]]
RunService.RenderStepped:Connect(function()
    local localPlayer = Players.LocalPlayer
    local anyHitboxOn = getgenv().HeadHitbox or getgenv().BodyHitbox or getgenv().MeleeHitbox

    -- Jogadores
    for _, v in next, Players:GetPlayers() do
        if v.Name ~= localPlayer.Name then
            pcall(function()
                if v.Character then
                    local teamOk = not getgenv().TeamCheck or (localPlayer.Team ~= v.Team)
                    if getgenv().HitboxStatus and anyHitboxOn and teamOk then
                        applyHitbox(v.Character, "Really black")
                    else
                        resetHitbox(v.Character)
                    end
                end
            end)
        end
    end

    -- NPCs
    local playerChars = getPlayerChars()
    for _, model in next, game.Workspace:GetDescendants() do
        if model:IsA("Model") and not playerChars[model] then
            pcall(function()
                local hrp = model:FindFirstChild("HumanoidRootPart")
                local hum = model:FindFirstChildOfClass("Humanoid")
                if hrp and hum then
                    if getgenv().NPCHitbox and anyHitboxOn then
                        applyHitbox(model, "Really red")
                    else
                        resetHitbox(model)
                    end
                end
            end)
        end
    end
end)

PlayerMovement:AddToggle('TPWalkToggle', {
    Text = 'TP Walk', Default = false,
    Callback = function(Value)
        getgenv().TPWalk = Value
        local hb = RunService.Heartbeat
        local lplr = Players.LocalPlayer
        while getgenv().TPWalk and hb:Wait() do
            local chr = lplr.Character
            local hum = chr and chr:FindFirstChildWhichIsA("Humanoid")
            if chr and hum and hum.Parent and hum.MoveDirection.Magnitude > 0 then
                if getgenv().TPSpeed and isNumber(getgenv().TPSpeed) then
                    chr:TranslateBy(hum.MoveDirection * tonumber(getgenv().TPSpeed))
                else
                    chr:TranslateBy(hum.MoveDirection)
                end
            end
        end
    end
})
